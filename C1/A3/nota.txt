def center_of_mass(n_rows, aisle_length, n_passengers, passenger_mass, seat_distribution):
    # Area de la sección de pasajeros
    passenger_area = (n_rows * 4 * 80 * 80) + (n_rows * aisle_length * 80)
    
    # Calcular el centro de masa de la sección de pasajeros
    passenger_x = 0
    passenger_y = 0
    total_mass = 0
    for i in range(n_rows):
        for j in range(4):
            if seat_distribution[i][j] != 0:
                # Coordenadas del asiento
                x = (i * 80) + (aisle_length / 2) + (j * 80)
                y = (j * 80) + (i * 80)
                # Añadir la masa del pasajero al total
                total_mass += passenger_mass[i][j]
                # Añadir la masa del pasajero al centro de masa
                passenger_x += x * passenger_mass[i][j]
                passenger_y += y * passenger_mass[i][j]
    # Calcular el centro de masa de la sección de pasajeros
    passenger_x /= total_mass
    passenger_y /= total_mass
    # Asumimos que el resto del avión tiene la misma masa y está equidistante
    # al centro de masa de la sección de pasajeros
    total_mass += (aisle_length * 80 * 80)
    x = (passenger_mass * passenger_x + total_mass * passenger_x) / total_mass
    y = (passenger_mass * passenger_y + total_mass * passenger_y) / total_mass
    
    return x, y
